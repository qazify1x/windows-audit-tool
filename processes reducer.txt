# --- Configuration & Safety Lists ---
$LogFile = "$env:USERPROFILE\Desktop\Optimizer_Log.txt"
$Global:RestoreList = @()
$CriticalProcesses = @("System", "Idle", "winlogon", "csrss", "lsass", "smss", "services", "svchost", "wininit", "explorer", "runtimebroker", "searchhost", "dwm", "fontdrvhost")

# --- Functions ---

function Write-Log {
    param([string]$Message)
    $Stamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    "[$Stamp] $Message" | Out-File -FilePath $LogFile -Append
}

function Get-ProcessCategory {
    param($Proc)
    $Name = $Proc.ProcessName.ToLower()
    $Company = $Proc.Company
    if ($CriticalProcesses -contains $Name) { return "CRITICAL (Red)" }
    if ($Company -match "Microsoft") { return "CAUTION (Yellow)" }
    return "SAFE (Green)"
}

function Refresh-ProcessList {
    $dataGridView.Rows.Clear()
    $Processes = Get-Process | Select-Object ProcessName, Id, CPU, WorkingSet, Path, Company | Sort-Object @{Expression={$_.Company -match "Microsoft"}}, ProcessName
    
    foreach ($P in $Processes) {
        $Risk = Get-ProcessCategory $P
        $Mem = [math]::Round($P.WorkingSet / 1MB, 2)
        $CpuVal = if ($P.CPU) { [math]::Round($P.CPU, 1) } else { 0 }
        
        $rowIdx = $dataGridView.Rows.Add($P.ProcessName, "$CpuVal %", "$Mem MB", $Risk, $P.Company, $P.Path)
        $row = $dataGridView.Rows[$rowIdx]

        switch ($Risk) {
            "CRITICAL (Red)" { 
                $row.Cells[3].Style.ForeColor = [System.Drawing.Color]::Red 
                $row.DefaultCellStyle.ForeColor = [System.Drawing.Color]::Gray
            }
            "CAUTION (Yellow)" { 
                $row.Cells[3].Style.ForeColor = [System.Drawing.Color]::Yellow 
            }
            "SAFE (Green)" { 
                $row.Cells[3].Style.ForeColor = [System.Drawing.Color]::LimeGreen 
                $row.Cells[0].Style.Font = New-Object System.Drawing.Font($dataGridView.Font, [System.Drawing.FontStyle]::Bold)
            }
        }
    }
}

function Optimize-System {
    $confirm = [System.Windows.Forms.MessageBox]::Show("Stop all SAFE (Green) processes? This is safe for Windows stability.", "Bulk Optimize", "YesNo", "Warning")
    if ($confirm -eq "No") { return }

    $Global:RestoreList = @() 
    $Procs = Get-Process | Select-Object ProcessName, Id, Company, Path

    foreach ($P in $Procs) {
        $Risk = Get-ProcessCategory $P
        if ($Risk -eq "SAFE (Green)" -and $P.ProcessName -notmatch "powershell|pwsh|code") {
            try {
                $Global:RestoreList += @{ Name = $P.ProcessName; Path = $P.Path }
                Stop-Process -Id $P.Id -Force -ErrorAction SilentlyContinue
                Write-Log "Bulk Stopped: $($P.ProcessName)"
            } catch {
                Write-Log "Failed to stop: $($P.ProcessName)"
            }
        }
    }
    Refresh-ProcessList
    [System.Windows.Forms.MessageBox]::Show("Optimization complete!", "Success")
}

function Restore-Changes {
    if ($Global:RestoreList.Count -eq 0) {
        [System.Windows.Forms.MessageBox]::Show("Nothing to restore in this session.", "Info")
        return
    }

    foreach ($Item in $Global:RestoreList) {
        if ($Item.Path -and (Test-Path $Item.Path)) {
            try {
                Start-Process $Item.Path -WindowStyle Minimized
                Write-Log "Restored: $($Item.Name)"
            } catch {
                Write-Log "Failed to restore: $($Item.Name)"
            }
        }
    }
    $Global:RestoreList = @()
    Refresh-ProcessList
    [System.Windows.Forms.MessageBox]::Show("Restore sequence finished.", "Restore")
}
